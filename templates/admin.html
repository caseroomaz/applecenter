import os import psycopg2 from psycopg2.extras import RealDictCursor from flask import Flask, render_template, request, redirect app = Flask(__name__) def get_db_connection(): url = os.environ.get('POSTGRES_URL') if url and url.startswith("postgres://"): url = url.replace("postgres://", "postgresql://", 1) return psycopg2.connect(url) @app.route("/") def index(): try: cat = request.args.get('cat') conn = get_db_connection() cur = conn.cursor(cursor_factory=RealDictCursor) # Kateqoriya seçilibsə ona görə, seçilməyibsə hamısını gətir if cat: cur.execute("SELECT * FROM products WHERE category ILIKE %s ORDER BY id DESC", (cat,)) else: cur.execute("SELECT * FROM products ORDER BY id DESC") products = cur.fetchall() # Navbar üçün kateqoriya siyahısı categories = ["iPhone", "iPad", "Mac", "Watch", "AirPods", "Aksesuar"] cur.close(); conn.close() return render_template("index.html", products=products, categories=categories) except Exception as e: return f"Ana səhifə xətası: {str(e)}" @app.route("/product/") def product_detail(id): try: conn = get_db_connection() cur = conn.cursor(cursor_factory=RealDictCursor) cur.execute("SELECT * FROM products WHERE id=%s", (id,)) product = cur.fetchone() cur.close(); conn.close() if not product: return "Məhsul tapılmadı", 404 # Çoxlu şəkil məntiqi: əsas şəkil + vergüllə ayrılmış digər şəkillər img_list = [product['image']] if product.get('extra_images'): img_list.extend(product['extra_images'].split(',')) product['images'] = img_list product['color_list'] = product.get('colors', "").split(',') if product.get('colors') else [] product['storage_list'] = product.get('storages', "").split(',') if product.get('storages') else [] return render_template("product.html", product=product) except Exception as e: return f"Məhsul səhifəsi xətası: {str(e)}" # Admin tərəfdə extra_images üçün yer ayırırıq @app.route("/admin/add", methods=["POST"]) def add(): conn = get_db_connection() cur = conn.cursor() cur.execute( "INSERT INTO products (name, price, image, category, colors, storages, extra_images) VALUES (%s,%s,%s,%s,%s,%s,%s)", (request.form['name'], request.form['price'], request.form['image'], request.form.get('category', 'iPhone'), request.form.get('colors', ''), request.form.get('storages', ''), request.form.get('extra_images', '')) ) conn.commit() cur.close(); conn.close() return redirect("/admin") application = app
